  <div id="modal" class="hidden fixed inset-0 bg-black/70 z-50 flex items-center justify-center p-4 backdrop-blur-sm">
    <div class="bg-white rounded-3xl w-full max-w-sm p-6">
      <h2 class="text-xl font-bold primary-green mb-4 text-center">İyiliği Paylaş</h2>
      <input type="file" id="fInp" accept="image/*" class="w-full mb-4 text-sm file:mr-4 file:py-2 file:px-4 file:rounded-full file:border-0 file:text-sm file:bg-emerald-50 file:text-emerald-700 hover:file:bg-emerald-100">
      <textarea id="cInp" placeholder="Bir açıklama yaz..." class="w-full border rounded-xl p-3 h-24 mb-4 outline-none focus:ring-2 focus:ring-emerald-500"></textarea>
      <div class="flex gap-3">
        <button onclick="closeModal()" class="flex-1 bg-gray-100 py-3 rounded-xl font-semibold text-gray-500">İptal</button>
        <button id="upBtn" onclick="upload()" class="flex-1 bg-primary text-white py-3 rounded-xl font-bold shadow-lg">Paylaş</button>
      </div>
    </div>

  <script type="module">
    import { initializeApp } from "https://www.gstatic.com/firebasejs/10.7.1/firebase-app.js";
    import { getAuth, GoogleAuthProvider, signInWithPopup, onAuthStateChanged } from "https://www.gstatic.com/firebasejs/10.7.1/firebase-auth.js";
    import { getFirestore, collection, addDoc, query, orderBy, onSnapshot, serverTimestamp } from "https://www.gstatic.com/firebasejs/10.7.1/firebase-firestore.js";
    import { getStorage, ref, uploadBytes, getDownloadURL } from "https://www.gstatic.com/firebasejs/10.7.1/firebase-storage.js";

    const firebaseConfig = {
      apiKey: "AIzaSyDZO3lct-xzXAx79lbqgMP7s6WRCirWIHk",
      authDomain: "hayravesile.firebaseapp.com",
      projectId: "hayravesile",
      storageBucket: "hayravesile.firebasestorage.app",
      messagingSenderId: "980073667334",
      appId: "1:980073667334:web:1f6260f0abf3bcc820dc87"
    };

    const app = initializeApp(firebaseConfig);
    const auth = getAuth(app);
    const db = getFirestore(app);
    const storage = getStorage(app);
    const provider = new GoogleAuthProvider();

    window.login = () => signInWithPopup(auth, provider);
    window.openModal = () => document.getElementById('modal').classList.remove('hidden');
    window.closeModal = () => document.getElementById('modal').classList.add('hidden');

    // Admin ve Giriş Kontrolü
    onAuthStateChanged(auth, user => {
      const area = document.getElementById('authArea');
      const btn = document.getElementById('adminBtn');
      if(user) {
        area.innerHTML = `<img src="${user.photoURL}" class="w-9 h-9 rounded-full border-2 border-emerald-600">`;
        if(user.email === "a.hamzabolat@gmail.com") btn.classList.remove('hidden');
      }
    });

    // Paylaşım Yapma
    window.upload = async () => {
      const file = document.getElementById('fInp').files[0];
      const cap = document.getElementById('cInp').value;
      const btn = document.getElementById('upBtn');

      if(!file || !cap) return alert("Lütfen fotoğraf ve yazı ekleyin.");
      
      btn.disabled = true; btn.innerText = "Yükleniyor...";

      try {
        const sRef = ref(storage, 'p/' + Date.now());
        const snap = await uploadBytes(sRef, file);
        const url = await getDownloadURL(snap.ref);

        await addDoc(collection(db, "posts"), {
          url, cap, user: auth.currentUser.displayName,
          uImg: auth.currentUser.photoURL, time: serverTimestamp()
        });
        
        closeModal();
        document.getElementById('fInp').value = "";
        document.getElementById('cInp').value = "";
      } catch (e) {
        alert("Hata: " + e.message);
      } finally {
        btn.disabled = false; btn.innerText = "Paylaş";
      }
    };

    // Akışı Getir
    const q = query(collection(db, "posts"), orderBy("time", "desc"));
    onSnapshot(q, snap => {
      const f = document.getElementById('feed');
      f.innerHTML = '';
      snap.forEach(d => {
        const p = d.data();
        f.innerHTML += `
          <div class="bg-white rounded-2xl shadow-sm border border-gray-100 overflow-hidden">
            <div class="p-3 flex items-center gap-2"><img src="${p.uImg}" class="w-6 h-6 rounded-full"><span class="text-xs font-bold">${p.user}</span></div>
            <img src="${p.url}" class="w-full">
            <div class="p-4 text-sm"><b>${p.user}</b> ${p.cap}</div>
          </div>`;
      });
      document.getElementById('loader').style.display = 'none';
    });
  </script>
